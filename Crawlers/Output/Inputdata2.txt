       function validateAvaSearch(selectDepartureDayId, selectDepartureMonthId, selectReturnDayId, selectReturnMonthId, selectDepartureCityId, selectArrivalCityId, tripTypeName)
        {
            var objSelectDepartureCity = document.getElementById(selectDepartureCityId);
            var objSelectArrivalCity = document.getElementById(selectArrivalCityId);
            
            if (objSelectDepartureCity.options[objSelectDepartureCity.selectedIndex].value == "" || objSelectDepartureCity.options[objSelectDepartureCity.selectedIndex].value == " ")
	        {
		        alert('Vælg afrejsepunkt');
		        return false;
	        }
            else if (!cityIsSelected(objSelectArrivalCity) || objSelectArrivalCity.options[objSelectArrivalCity.selectedIndex].value == "selectDestination" || objSelectArrivalCity.options[objSelectArrivalCity.selectedIndex].value == "" || objSelectArrivalCity.options[objSelectArrivalCity.selectedIndex].value == " ")
	        {
		        alert('Vælg destination');
		        return false;
	        }
	        else
	        {
		        return validateDates(selectDepartureDayId, selectDepartureMonthId, selectReturnDayId, selectReturnMonthId, tripTypeName, false);
	        }
        }

        function validateDates(selectDepartureDayId, selectDepartureMonthId, selectReturnDayId, selectReturnMonthId, tripTypeName, hideWarning)
        {
            var objSelectDepartureDay = document.getElementById(selectDepartureDayId);
            var objSelectDepartureMonth = document.getElementById(selectDepartureMonthId);
            var objSelectReturnDay = document.getElementById(selectReturnDayId);
            var objSelectReturnMonth = document.getElementById(selectReturnMonthId);
	        var errMesg = "";
            
            var departureDate = extractDate(objSelectDepartureDay, objSelectDepartureMonth);
            if (!isValidDate(departureDate))
                errMesg = errMesg + 'Afgangsdatoen er ikke gyldig\n';
            
	        if ($('input:radio[name=' + tripTypeName + ']:checked').val() == '2') {
                var returnDate = extractDate(objSelectReturnDay, objSelectReturnMonth);
                if (!isValidDate(returnDate))
                    errMesg = errMesg + 'Returdatoen er ikke gyldig\n';
                
		        if(returnDate < departureDate)
			        errMesg = errMesg + 'Udgående flyvning kan ikke ligge senere end returflyvning!\n';
	        }

	        var today = new Date();
	        if (new Date(departureDate.getFullYear(), departureDate.getMonth(), departureDate.getDate(), 23, 59, 59, 999) < today)
		        errMesg = errMesg + 'Afgangsdatoen er ikke gyldig\n';

	        if (errMesg.length > 0) {
		        if (!hideWarning)
			        alert(errMesg);
		        return false;
	        }
	        else 
		        return true;
        }

        function farecalValidateDate(selectDepartureMonthId, selectReturnMonthId, hideWarning)
		{
            var objSelectDepartureMonth = document.getElementById(selectDepartureMonthId);
            var objSelectReturnMonth = document.getElementById(selectReturnMonthId);
			var errMesg = "";
                
            var returnMonth = objSelectReturnMonth.options[objSelectReturnMonth.selectedIndex].value;
            var departureMonth = objSelectDepartureMonth.options[objSelectDepartureMonth.selectedIndex].value;
            if (returnMonth != '' && (parseInt(returnMonth, 10) < parseInt(departureMonth,10))) {
                errMesg = errMesg + 'Udgående flyvning kan ikke ligge senere end returflyvning!\n';
			}
			if (errMesg.length > 0) {
				if (!hideWarning)
					alert(errMesg);
				return false;
			}
			else 
				return true;
		}
        
        function farecalValidate(selectDepartureMonthId, selectReturnMonthId, selectDepartureCityId, selectArrivalCityId)
		{
            var objSelectDepartureCity = document.getElementById(selectDepartureCityId);
            var objSelectArrivalCity = document.getElementById(selectArrivalCityId);
            var objSelectDepartureMonth = document.getElementById(selectDepartureMonthId);
            var objSelectReturnMonth = document.getElementById(selectReturnMonthId);  
            if (objSelectDepartureCity.options[objSelectDepartureCity.selectedIndex].value == "" || objSelectDepartureCity.options[objSelectDepartureCity.selectedIndex].value == " ")
			{
				alert('Vælg afrejsepunkt');
				return false;
			}
            else if (!cityIsSelected(objSelectArrivalCity) || objSelectArrivalCity.options[objSelectArrivalCity.selectedIndex].value == "" || objSelectArrivalCity.options[objSelectArrivalCity.selectedIndex].value == " ")
			{
				alert('Vælg destination');
				return false;
			}
			else
			{
				return farecalValidateDate(selectDepartureMonthId, selectReturnMonthId, false);
			}
		}

        function getFareCalendar(selectDepartureMonthId, selectReturnMonthId, selectDepartureCityId, selectArrivalCityId, selectAdultsId, selectChildrenId, selectInfantsId, tripType, selectedDepartureDay, selectedReturnDay)
		{
            var objSelectDepartureCity = document.getElementById(selectDepartureCityId);
            var objSelectArrivalCity = document.getElementById(selectArrivalCityId);
            var objSelectDepartureMonth = document.getElementById(selectDepartureMonthId);
            var objSelectReturnMonth = document.getElementById(selectReturnMonthId);                
            var objSelectAdults = document.getElementById(selectAdultsId);
            var objSelectChildren = document.getElementById(selectChildrenId);
            var objSelectInfants = document.getElementById(selectInfantsId); 
            var numAdults = 1;
            var numChildren = 0;
            var numInfants = 0;            

			if (farecalValidate(selectDepartureMonthId, selectReturnMonthId, selectDepartureCityId, selectArrivalCityId))
            {
                var valDCity = objSelectDepartureCity.value;
                var valACity = objSelectArrivalCity.value;
                var valDMonth = objSelectDepartureMonth.value;
                var valRMonth = objSelectReturnMonth.value;

                var valTripType;
                if(typeof(tripType)=='undefined' || tripType == null)
                {
			        if (objSelectReturnMonth.selectedIndex != 0) {
                       valTripType = 2; 
                    }
                    else {
                        valTripType = 1;
                    }
                }
                else
                {
                    valTripType = tripType;
                }

                if (objSelectAdults != null && '' + objSelectAdults != 'undefined'){
                    numAdults = objSelectAdults.value;
                    numChildren = objSelectChildren.value;
                    numInfants = objSelectInfants.value;
                }
                
                var query = '?D_City=' + valDCity+ '&A_City=' + valACity + '&TripType=' + valTripType + '&D_Month=' + valDMonth + '&R_Month=' + valRMonth + '&AdultCount=' + numAdults + '&ChildCount=' + numChildren + '&InfantCount=' + numInfants;
                if(objSelectAdults != null && '' + objSelectAdults != 'undefined' && objSelectAdults.value == 'x')
                {
                    if(typeof(selectedDepartureDay)!='undefined' && selectedDepartureDay != null)
                    {
                        query += '&D_Day=' + selectedDepartureDay.value;
                    }
                    if(typeof(selectedReturnDay)!='undefined' && selectedReturnDay != null)
                    {
                        query += '&R_Day=' + selectedReturnDay.value;
                    }
                    window.location.href = '/ssl/dk/processpages/grouptravel/grouptravelavaday/' + query;
                }
                else{
                    if(typeof(selectedDepartureDay)!='undefined' && selectedDepartureDay != null)
                    {
                        query += '&D_Day=1&D_SelectedDay=' + selectedDepartureDay.value;
                    }
                    if(typeof(selectedReturnDay)!='undefined' && selectedReturnDay != null)
                    {
                        query += '&R_Day=1&R_SelectedDay=' + selectedReturnDay.value;
                    }
                    window.location.href = '/dk/fly/lavpris/' + query;
                }        
            }
		}
